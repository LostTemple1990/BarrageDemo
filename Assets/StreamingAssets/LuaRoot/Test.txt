--测试虚化子弹
CustomizedTable.TestUnrealBullet = {}
CustomizedTable.TestUnrealBullet.Init = function(bullet,velocity,angle,isAimToPlayer)
	lib.AddBulletTask(bullet,function()
		lib.SetBulletStraightParas(bullet,velocity,angle,isAimToPlayer,0,0)
		lib.SetBulletToUnrealState(bullet,60)
		if coroutine.yield(60) == false then return end
	end)
end
